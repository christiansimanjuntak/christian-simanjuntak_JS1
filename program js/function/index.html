function hitungFaktorial(n) { if (n < 0) { return "Faktorial hanya didefinisikan untuk bilangan non-negatif."; } else if (n === 0 || n === 1) { return 1; } else { return n * hitungFaktorial(n - 1); } } const angka = 5; const faktorial = hitungFaktorial(angka); console.log(`Faktorial dari ${angka} adalah: ${faktorial}`);
